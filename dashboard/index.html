-- Ultimate 3-source JOIN across YouTube, Twitter & Discord
SELECT
  y.title,
  t.like_count,
  t.impression_count,
  d.content,
  d.author_username
FROM youtube.videos y,
     twitter.tweets t,
     discord.messages d
LIMIT 5;
